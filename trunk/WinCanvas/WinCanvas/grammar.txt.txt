namespace CanvasParser
{
    
    grammar CanvasParser
    {
        token TKOPENP       = '(';
        token TKCLOSEP      = ')';
        token TKCOMA        = ',';
        token TKRGB         = "rgb";
        token TKRGBA        = "rgba";
        token TKNUMBER      = ("0".."9")+;
        token TKFNUMBER     = ("0".."9")+ ("." "0".."9"+)?;
        interleave BLANKS     = ((" " | "\n" | "\t" | "\r")+ );
    }
}
